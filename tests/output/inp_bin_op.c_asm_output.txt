section .data
__main_size: equ 64
section .text
global main
main:
push rbp
mov rbp, rsp
sub rsp, __main_size
mov qword [rsp+0], 8
add [rsp+0], rsp
mov qword [rsp+16], 24
add [rsp+16], rsp
mov rax, rsp
add rax, 0
mov [rsp+32], rax
mov rax, rsp
add rax, 16
mov [rsp+40], rax
mov rax, [rsp+0]
mov rbx, [rsp+16]
add rax, rbx
mov rbx, [rsp+32]
mov [rbx], rax
mov [rsp+48], rax
mov rax, 0
mov rsp, rbp
pop rbp
ret
mov rsp, rbp
pop rbp
ret
extern exit
global _start
_start:
call main
mov rdi, 0
call exit
